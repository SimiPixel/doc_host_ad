import copy 
from .utils import convert_to_timestamp, padd_df, sort_df
from .anomaly.base import Base_AnomalyDetection
from .forecast.base import Base_Forecast

class Constructor(object):

    '''
    Object that merges and performs anomaly detection and forecast.
    So, we find the a-normal object by comparing its values to the other objects' values at the *same time and the same comparison points*.

    Args:
        df (pd.DataFrame): The input data.

        time_col (string): The name of the column in the df that gives the timestamp of the event.

        obj_col (string): The name of the column in the df that identifies the object to find.

        comp_col (list): The name of the column which is used as comparison point.

        value_col (list): List of column-names which are compared.

        value_agg (list): List of agg-functions.
            default = _*['mean']

        bin_size_time (string): The bin size of time aggregation.
            default = '1H'

        n_bins_past (int): Number of time bins in the past to consider for the forecast.
            default = 4

        n_bins_pred (int): Number of time bins to predict.
            default = 2

    Methods:
        .set_anomaly_detection_model(ad, ad_kwargs):
            ad (ad.anomaly.base.Base_AnomalyDetection): The model.
            ad_kwargs: The parameters of the model.

        .set_forecast_model(forecast, forecast_kwargs):
            forecast (ad.forecast.base.Base_Forecast): The model.
            forecast_kwargs: The parameters of the model.

        .run():
            Returns (pd.DataFrame): Dataframe with anomalyScores.

    Example Usage:
        Input-df:
                    time            obj         comp       value
                0   (pd.Timestamp)  (int)       (int)      (float)
                1   (pd.Timestamp)  (int)       (int)      (float)
                ...
                _   (pd.Timestamp)  (int)       (int)      (float)

        from ad.constructor import Constructor
        from ad.anomaly import VoteDeviation

        constr = Constructor(
            df,
            time_col = 'time',
            obj_col = 'obj', 
            comp_col = 'comp',
            value_col= ['value'],
        )

        constr.set_anomaly_detection_model(VoteDeviation)

        # Optional 
        # from ad.forecast import GaussianProcess
        # constr.set_forecast_model(GaussianProcess)

        df_anomaly = constr.run()

    '''

    def __init__(
        self, df, time_col, obj_col, comp_col, value_col, 
        value_agg = None,
        bin_size_time = '1H',
        n_bins_past = 4,
        n_bins_pred = 2
        ):

        self.df = copy.copy(df)

        self.time = time_col
        self.obj = obj_col
        self.comp = comp_col
        self.value = value_col
        self.value_agg = value_agg
        self.time_bin = bin_size_time
        self.n_bins_past = n_bins_past
        self.n_bins_pred = n_bins_pred

        self.ad = None 
        self.forecast = None

        if self.value_agg is None:
            self.value_agg = len(self.value)*['mean']

        # Check if column names are present in df
        if not set([time_col, obj_col, comp_col] + value_col).issubset(set(self.df.columns.to_list())):
            raise ValueError('The chosen column names are not present in the df.')

        # Swap columns into right order
        self.df = self.df.reindex(columns = [self.time]+[self.obj]+[self.comp]+self.value)

        # Convert timestamp
        convert_to_timestamp(self.df, self.time)

        # Grouping and aggregating
        self.df = self.df.groupby(
                [
                    pd.Grouper(
                    key = self.time,
                    freq = self.time_bin
                    ),
                    self.obj,
                    self.comp
                ]
            ).agg(dict(zip(self.value, self.value_agg))).reset_index()

        # Padd missing values / insert NaN's
        self.df = padd_df(self.df)

        # Sort df
        self.df = sort_df(self.df)


    def set_anomaly_detection_model(self, ad, **ad_args):

        self.AD = ad 
        self.ad_args = ad_args

        self.ad = ad(self.df, **ad_args)
        if not isinstance(self.ad, Base_AnomalyDetection):
            raise TypeError('Provided anomaly detection model is not as expected.')

    def set_forecast_model(self, forecast, **forecast_args):
        
        # Check if provided time > n_bins_past
        if self.df[self.time].unique().size < self.n_bins_past:
            raise ValueError(
                f'The input dataframe does not span enough time\
                 , so {self.n_bins_pred}x{self.time_bin}.')

        # Truncate df to only include the n_bins_pred timestamps
        df_trunc = self.df[self.df[self.time] >= self.df[self.time].unique()[-self.n_bins_past]]

        # Number of rows in df 
        idx_of_last_row = len(self.df.index)

        self.forecast = forecast(
            df = df_trunc, 
            n_bins_past = self.n_bins_past, 
            n_bins_pred = self.n_bins_pred,
            idx_of_first_row = idx_of_last_row +1,  
            **forecast_args)
            
        if not isinstance(self.forecast, Base_Forecast):
            raise TypeError('Provided forecast model is not as expected.')

    def _cleanup(self, df_anomaly):
        timestamps = df_anomaly[self.time].unique()
        start = timestamps[0]
        end =  timestamps[-1]

        # All timestamps between start and end
        multi_index = pd.MultiIndex.from_product([pd.date_range(start = start, end = end, freq= self.time_bin), df_anomaly[self.obj].unique()], names = [self.time, self.obj])

        df_NaN = pd.DataFrame(data = {}, index = df_anomaly.columns[:2].to_list(), columns = df_anomaly.columns[2:3].to_list()).reindex(multi_index)

        # Merge with existing, to fill in all missing time stamps with NaN values, plotly needs this
        df_anomaly_multi = df_anomaly.set_index(df_anomaly.columns[:2].to_list())
        df_app = df_anomaly_multi.append(df_NaN)

        return df_app[~df_app.index.duplicated(keep='first')].reset_index().sort_values(df_anomaly.columns[:2].to_list())


    def run(self):

        # Do forecast if it was set previously
        if self.forecast is not None:
            df_forecast = self.forecast.do_forecast()

            # Append forecast df to existing df
            df_extended = self.df.append(df_forecast)

            # Re-set anomaly detection with longer df if forecast was set previously 
            self.ad = self.AD(df_extended, **self.ad_args)


        # Do anomaly detection
        df_anomaly = self.ad.calc_anomaly()

        # Padd all missing times with NaN
        df_anomaly = self._cleanup(df_anomaly)

        return df_anomaly
